def knuth(n,a,b): # returns a ↑ⁿ b
    if n == 1:
        return a ** b
    elif b == 0:
        return 1
    else:
        return knuth(n-1,a,knuth(n,a,b-1))

def g(n):
    if n == 1:
        return knuth(4,3,3) # returns 3 ↑↑↑↑ 3
    else:
        return knuth(g(n-1),3,3)

print(g(64)) # should print Graham's number, if your PC can handle it
